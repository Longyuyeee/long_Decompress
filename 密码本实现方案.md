### 2.2 加密模块 (crypto.py)

```python
import os
import base64
from typing import Tuple
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.exceptions import InvalidTag

class CryptoManager:
    """加密管理器"""

    def __init__(self, salt_size: int = 16, iterations: int = 100000):
        self.salt_size = salt_size
        self.iterations = iterations

    def derive_key(self, password: str, salt: bytes = None) -> Tuple[bytes, bytes]:
        """派生加密密钥"""
        if salt is None:
            salt = os.urandom(self.salt_size)

        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,  # AES-256密钥长度
            salt=salt,
            iterations=self.iterations
        )

        key = kdf.derive(password.encode('utf-8'))
        return key, salt

    def encrypt(self, plaintext: str, password: str) -> str:
        """加密数据"""
        # 生成随机盐和IV
        salt = os.urandom(self.salt_size)
        iv = os.urandom(12)  # GCM推荐12字节IV

        # 派生密钥
        key, _ = self.derive_key(password, salt)

        # 加密
        aesgcm = AESGCM(key)
        ciphertext = aesgcm.encrypt(iv, plaintext.encode('utf-8'), None)

        # 组合数据: salt + iv + ciphertext
        combined = salt + iv + ciphertext
        return base64.b64encode(combined).decode('utf-8')

    def decrypt(self, encrypted_data: str, password: str) -> str:
        """解密数据"""
        # 解码base64
        combined = base64.b64decode(encrypted_data)

        # 分离各部分
        salt = combined[:self.salt_size]
        iv = combined[self.salt_size:self.salt_size + 12]
        ciphertext = combined[self.salt_size + 12:]

        # 派生密钥
        key, _ = self.derive_key(password, salt)

        # 解密
        aesgcm = AESGCM(key)
        try:
            plaintext = aesgcm.decrypt(iv, ciphertext, None)
            return plaintext.decode('utf-8')
        except InvalidTag:
            raise ValueError("解密失败：密码错误或数据损坏")
```

### 2.3 数据库模块 (database.py)

```python
import sqlite3
import json
from datetime import datetime
from typing import List, Optional, Dict, Any
from pathlib import Path

class PasswordDatabase:
    """密码数据库管理器"""

    def __init__(self, db_path: str, password: str = None):
        self.db_path = Path(db_path)
        self.password = password
        self.connection = None

    def connect(self):
        """连接数据库"""
        if self.password:
            # 使用SQLCipher加密
            self.connection = sqlite3.connect(str(self.db_path))
            self.connection.execute(f"PRAGMA key='{self.password}'")
        else:
            self.connection = sqlite3.connect(str(self.db_path))

        # 启用外键和WAL模式
        self.connection.execute("PRAGMA foreign_keys = ON")
        self.connection.execute("PRAGMA journal_mode = WAL")

        # 创建表
        self._create_tables()

    def _create_tables(self):
        """创建数据库表"""
        scripts = [
            """
            CREATE TABLE IF NOT EXISTS passwords (
                id TEXT PRIMARY KEY,
                filename TEXT NOT NULL,
                encrypted_password TEXT NOT NULL,
                file_hash TEXT,
                tags_json TEXT DEFAULT '[]',
                category TEXT,
                strength INTEGER DEFAULT 3,
                usage_count INTEGER DEFAULT 0,
                success_count INTEGER DEFAULT 0,
                last_used TIMESTAMP,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                source TEXT DEFAULT 'manual',
                notes TEXT,
                metadata_json TEXT DEFAULT '{}'
            )
            """,
            """
            CREATE TABLE IF NOT EXISTS tags (
                id TEXT PRIMARY KEY,
                name TEXT UNIQUE NOT NULL,
                color TEXT DEFAULT '#3498db',
                description TEXT,
                priority INTEGER DEFAULT 0,
                is_system BOOLEAN DEFAULT 0,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
            """,
            """
            CREATE TABLE IF NOT EXISTS usage_history (
                id TEXT PRIMARY KEY,
                password_id TEXT,
                filename TEXT NOT NULL,
                success BOOLEAN NOT NULL,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                details_json TEXT DEFAULT '{}',
                FOREIGN KEY (password_id) REFERENCES passwords(id) ON DELETE CASCADE
            )
            """,
            """
            CREATE INDEX IF NOT EXISTS idx_passwords_filename ON passwords(filename);
            CREATE INDEX IF NOT EXISTS idx_passwords_filehash ON passwords(file_hash);
            CREATE INDEX IF NOT EXISTS idx_passwords_category ON passwords(category);
            CREATE INDEX IF NOT EXISTS idx_usage_history_password ON usage_history(password_id);
            CREATE INDEX IF NOT EXISTS idx_usage_history_timestamp ON usage_history(timestamp);
            """
        ]

        cursor = self.connection.cursor()
        for script in scripts:
            cursor.execute(script)
        self.connection.commit()

    def add_password(self, record: 'PasswordRecord') -> str:
        """添加密码记录"""
        cursor = self.connection.cursor()

        cursor.execute("""
            INSERT INTO passwords (
                id, filename, encrypted_password, file_hash, tags_json,
                category, strength, usage_count, success_count, last_used,
                source, notes, metadata_json
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            record.id,
            record.filename,
            record.encrypted_password,
            record.file_hash,
            json.dumps(record.tags),
            record.category,
            record.strength,
            record.usage_count,
            record.success_count,
            record.last_used.isoformat() if record.last_used else None,
            record.source,
            record.notes,
            json.dumps(record.metadata)
        ))

        self.connection.commit()
        return record.id

    def find_password_by_filename(self, filename: str, limit: int = 10) -> List[Dict]:
        """根据文件名查找密码"""
        cursor = self.connection.cursor()

        cursor.execute("""
            SELECT * FROM passwords
            WHERE filename LIKE ?
            ORDER BY usage_count DESC, last_used DESC
            LIMIT ?
        """, (f"%{filename}%", limit))

        rows = cursor.fetchall()
        return self._rows_to_dicts(cursor, rows)

    def record_usage(self, history: 'UsageHistory'):
        """记录使用历史"""
        cursor = self.connection.cursor()

        cursor.execute("""
            INSERT INTO usage_history (id, password_id, filename, success, details_json)
            VALUES (?, ?, ?, ?, ?)
        """, (
            history.id,
            history.password_id,
            history.filename,
            history.success,
            json.dumps(history.details)
        ))

        # 更新密码使用统计
        if history.password_id:
            cursor.execute("""
                UPDATE passwords
                SET usage_count = usage_count + 1,
                    success_count = success_count + ?,
                    last_used = ?
                WHERE id = ?
            """, (
                1 if history.success else 0,
                history.timestamp.isoformat(),
                history.password_id
            ))

        self.connection.commit()

    def _rows_to_dicts(self, cursor, rows):
        """将查询结果转换为字典列表"""
        columns = [column[0] for column in cursor.description]
        result = []

        for row in rows:
            row_dict = dict(zip(columns, row))

            # 解析JSON字段
            if 'tags_json' in row_dict:
                row_dict['tags'] = json.loads(row_dict.pop('tags_json'))
            if 'metadata_json' in row_dict:
                row_dict['metadata'] = json.loads(row_dict.pop('metadata_json'))

            result.append(row_dict)

        return result

    def close(self):
        """关闭数据库连接"""
        if self.connection:
            self.connection.close()
```

### 2.4 智能匹配算法 (matcher.py)

```python
import re
from typing import List, Dict, Any
from difflib import SequenceMatcher
from collections import defaultdict

class SmartPasswordMatcher:
    """智能密码匹配器"""

    def __init__(self, database: 'PasswordDatabase'):
        self.db = database
        self.cache = {}

        # 预定义模式
        self.patterns = {
            'date': r'\d{4}[-_]\d{2}[-_]\d{2}|\d{8}',
            'version': r'v\d+(\.\d+)*',
            'serial': r'[A-Za-z]+\d+',
            'common_words': ['password', 'pass', 'key', 'secret', 'code']
        }

    def find_passwords(self, filename: str, context: Dict[str, Any] = None, limit: int = 10) -> List[Dict]:
        """查找匹配的密码"""
        if not context:
            context = {}

        # 1. 精确匹配
        exact_matches = self._find_exact_matches(filename)

        # 2. 模糊匹配
        fuzzy_matches = self._find_fuzzy_matches(filename)

        # 3. 模式匹配
        pattern_matches = self._find_pattern_matches(filename, context)

        # 4. 组合所有匹配
        all_matches = self._combine_matches(exact_matches, fuzzy_matches, pattern_matches)

        # 5. 计算优先级并排序
        prioritized = self._prioritize_matches(all_matches, filename, context)

        return prioritized[:limit]

    def _find_exact_matches(self, filename: str) -> List[Dict]:
        """精确匹配"""
        # 按完整文件名匹配
        matches = self.db.find_password_by_filename(filename, limit=5)

        # 按文件哈希匹配（如果有）
        file_hash = self._calculate_file_hash(filename)
        if file_hash:
            hash_matches = self.db.find_password_by_file_hash(file_hash)
            matches.extend(hash_matches)

        return matches

    def _find_fuzzy_matches(self, filename: str) -> List[Dict]:
        """模糊匹配"""
        # 获取所有密码记录进行相似度计算
        all_passwords = self.db.get_all_passwords(limit=100)
        fuzzy_matches = []

        for password in all_passwords:
            similarity = self._calculate_similarity(filename, password['filename'])
            if similarity > 0.6:  # 相似度阈值
                password['similarity'] = similarity
                fuzzy_matches.append(password)

        return fuzzy_matches

    def _find_pattern_matches(self, filename: str, context: Dict) -> List[Dict]:
        """模式匹配"""
        matches = []

        # 提取文件名中的模式
        patterns_found = self._extract_patterns(filename)

        # 查找具有相似模式的密码
        for pattern in patterns_found:
            pattern_matches = self.db.find_passwords_by_pattern(pattern)
            for match in pattern_matches:
                match['pattern_match'] = pattern
                matches.append(match)

        return matches

    def _calculate_similarity(self, str1: str, str2: str) -> float:
        """计算字符串相似度"""
        # 1. 编辑距离相似度
        seq_matcher = SequenceMatcher(None, str1.lower(), str2.lower())
        edit_similarity = seq_matcher.ratio()

        # 2. 公共词相似度
        words1 = set(re.findall(r'\b\w+\b', str1.lower()))
        words2 = set(re.findall(r'\b\w+\b', str2.lower()))

        if words1 and words2:
            common_words = words1.intersection(words2)
            word_similarity = len(common_words) / max(len(words1), len(words2))
        else:
            word_similarity = 0

        # 3. 前缀/后缀相似度
        prefix_len = 0
        for i in range(min(len(str1), len(str2))):
            if str1[i].lower() == str2[i].lower():
                prefix_len += 1
            else:
                break

        prefix_similarity = prefix_len / max(len(str1), len(str2))

        # 加权组合
        return (edit_similarity * 0.5 +
                word_similarity * 0.3 +
                prefix_similarity * 0.2)

    def _extract_patterns(self, filename: str) -> List[str]:
        """从文件名中提取模式"""
        patterns = []

        # 日期模式
        date_matches = re.findall(self.patterns['date'], filename)
        patterns.extend(date_matches)

        # 版本模式
        version_matches = re.findall(self.patterns['version'], filename)
        patterns.extend(version_matches)

        # 序列号模式
        serial_matches = re.findall(self.patterns['serial'], filename)
        patterns.extend(serial_matches)

        # 常见单词
        for word in self.patterns['common_words']:
            if word in filename.lower():
                patterns.append(word)

        return patterns

    def _combine_matches(self, *match_lists):
        """合并匹配结果，去重"""
        seen_ids = set()
        combined = []

        for match_list in match_lists:
            for match in match_list:
                if match['id'] not in seen_ids:
                    seen_ids.add(match['id'])
                    combined.append(match)

        return combined

    def _prioritize_matches(self, matches: List[Dict], filename: str, context: Dict) -> List[Dict]:
        """计算匹配优先级并排序"""
        for match in matches:
            # 基础优先级分数
            priority = 0.0

            # 使用频率权重
            usage_score = min(match.get('usage_count', 0) / 100, 1.0) * 0.3
            priority += usage_score

            # 成功率权重
            success_rate = match.get('success_count', 0) / max(match.get('usage_count', 1), 1)
            success_score = success_rate * 0.2
            priority += success_score

            # 相似度权重（如果有）
            if 'similarity' in match:
                similarity_score = match['similarity'] * 0.3
                priority += similarity_score
            else:
                # 精确匹配有额外加分
                if match['filename'] == filename:
                    priority += 0.4

            # 时间衰减
            if match.get('last_used'):
                # 简单的时间衰减：最近使用加分
                # 实际实现中应该使用更复杂的时间衰减函数
                priority += 0.1

            match['priority'] = priority

        # 按优先级排序
        return sorted(matches, key=lambda x: x.get('priority', 0), reverse=True)
```

### 2.5 密码生成器 (generator.py)

```python
import random
import string
import re
from datetime import datetime
from typing import List, Optional

class PasswordGenerator:
    """密码生成器"""

    def __init__(self):
        # 字符集定义
        self.lowercase = string.ascii_lowercase
        self.uppercase = string.ascii_uppercase
        self.digits = string.digits
        self.symbols = "!@#$%^&*()_+-=[]{}|;:,.<>?"

        # 常见密码模式
        self.common_patterns = [
            "{word}{year}",
            "{word}{symbol}{year}",
            "{word}{digit}{digit}",
            "{word}{symbol}{digit}{digit}",
            "{year}{word}",
            "{word}{month}{day}"
        ]

    def generate_random(self, length: int = 12, complexity: str = "medium") -> str:
        """生成随机密码"""
        if complexity == "weak":
            charset = self.lowercase + self.digits
        elif complexity == "medium":
            charset = self.lowercase + self.uppercase + self.digits
        elif complexity == "strong":
            charset = self.lowercase + self.uppercase + self.digits + self.symbols
        else:
            charset = self.lowercase + self.uppercase + self.digits + self.symbols

        # 确保至少包含每种类型（对于中等和强复杂度）
        password_chars = []

        if complexity in ["medium", "strong"]:
            password_chars.append(random.choice(self.lowercase))
            password_chars.append(random.choice(self.uppercase))
            password_chars.append(random.choice(self.digits))

        if complexity == "strong":
            password_chars.append(random.choice(self.symbols))

        # 填充剩余长度
        remaining = length - len(password_chars)
        if remaining > 0:
            password_chars.extend(random.choices(charset, k=remaining))

        # 随机打乱
        random.shuffle(password_chars)
        return ''.join(password_chars)

    def generate_from_filename(self, filename: str) -> List[str]:
        """基于文件名生成密码"""
        suggestions = []

        # 提取文件名中的关键词
        keywords = self._extract_keywords(filename)

        # 当前年份
        current_year = datetime.now().year
        short_year = str(current_year)[-2:]

        # 常见日期
        common_dates = [
            str(current_year),
            short_year,
            "1234",
            "2024",
            "2023",
            "2022"
        ]

        # 常见符号
        common_symbols = ["!", "@", "#", "$", "%", "&", "*", "_", "-", "="]

        # 生成密码变体
        for keyword in keywords[:3]:  # 最多使用前3个关键词
            for pattern in self.common_patterns:
                for date in common_dates[:2]:
                    for symbol in common_symbols[:2]:
                        password = pattern.format(
                            word=keyword,
                            year=date,
                            symbol=symbol,
                            digit=random.choice(self.digits),
                            month=str(datetime.now().month).zfill(2),
                            day=str(datetime.now().day).zfill(2)
                        )

                        # 添加一些随机变体
                        variations = self._generate_variations(password)
                        suggestions.extend(variations)

        # 去重并限制数量
        unique_suggestions = list(set(suggestions))
        return unique_suggestions[:20]  # 最多返回20个建议

    def _extract_keywords(self, filename: str) -> List[str]:
        """从文件名中提取关键词"""
        # 移除扩展名
        name_without_ext = filename.rsplit('.', 1)[0] if '.' in filename else filename

        # 分割单词（支持驼峰、下划线、连字符等）
        words = re.split(r'[_\-\s\.]+', name_without_ext)

        # 过滤短单词和常见停用词
        stop_words = {'the', 'and', 'for', 'with', 'from', 'this', 'that'}
        keywords = [
            word.lower()
            for word in words
            if len(word) > 2 and word.lower() not in stop_words
        ]

        return keywords

    def _generate_variations(self, password: str) -> List[str]:
        """生成密码变体"""
        variations = [password]

        # 大小写变体
        variations.append(password.lower())
        variations.append(password.upper())

        # 首字母大写
        if password:
            variations.append(password[0].upper() + password[1:].lower())

        # 替换常见字符
        replacements = {
            'a': ['@', '4'],
            'e': ['3'],
            'i': ['1', '!'],
            'o': ['0'],
            's': ['$', '5'],
            't': ['7']
        }

        for char, replacements_list in replacements.items():
            for replacement in replacements_list:
                if char in password.lower():
                    variations.append(password.lower().replace(char, replacement))

        return list(set(variations))
```
