#### 2.2 模糊匹配算法
```rust
fn fuzzy_match(query: &str, target: &str, threshold: f32) -> bool {
    if query.is_empty() || target.is_empty() {
        return false;
    }

    // 使用改进的编辑距离算法
    let distance = improved_edit_distance(query, target);
    let max_len = query.len().max(target.len()) as f32;

    // 计算相似度
    let similarity = 1.0 - (distance as f32 / max_len);

    similarity >= threshold
}

fn improved_edit_distance(a: &str, b: &str) -> usize {
    // 使用动态规划计算编辑距离
    let mut dp = vec![vec![0; b.len() + 1]; a.len() + 1];

    for i in 0..=a.len() {
        dp[i][0] = i;
    }
    for j in 0..=b.len() {
        dp[0][j] = j;
    }

    for i in 1..=a.len() {
        for j in 1..=b.len() {
            let cost = if a.chars().nth(i - 1) == b.chars().nth(j - 1) {
                0
            } else {
                1
            };

            dp[i][j] = (dp[i - 1][j] + 1)  // 删除
                .min(dp[i][j - 1] + 1)     // 插入
                .min(dp[i - 1][j - 1] + cost); // 替换
        }
    }

    dp[a.len()][b.len()]
}
```

### 3. 相关性排序算法

#### 3.1 BM25算法实现
```rust
struct BM25Scorer {
    avg_doc_length: f32,
    k1: f32,
    b: f32,
}

impl BM25Scorer {
    fn score(&self, term_freq: usize, doc_length: usize, doc_freq: usize, total_docs: usize) -> f32 {
        let idf = ((total_docs as f32 - doc_freq as f32 + 0.5) /
                   (doc_freq as f32 + 0.5) + 1.0).ln();

        let tf = term_freq as f32 * (self.k1 + 1.0) /
                (term_freq as f32 + self.k1 * (1.0 - self.b +
                 self.b * doc_length as f32 / self.avg_doc_length));

        idf * tf
    }
}
```

struct QueryCache {
    cache: LruCache<String, CachedResult>,
    ttl: Duration,
}

impl QueryCache {
    fn get_or_compute(&mut self, query_key: &str, compute: impl Fn() -> QueryResult) -> QueryResult {
        if let Some(cached) = self.cache.get(query_key) {
            if cached.timestamp.elapsed() < self.ttl {
                return cached.result.clone();
            }
        }

        let result = compute();
        self.cache.put(
            query_key.to_string(),
            CachedResult {
                result: result.clone(),
                timestamp: Instant::now(),
            }
        );

        result
    }
}

## 安全查询实现

### 1. 加密字段查询策略

由于敏感字段是加密存储的，查询需要特殊处理：

```rust
struct SecureQueryExecutor {
    encryption_service: Arc<EncryptionService>,
    search_index: SecureSearchIndex,
}

impl SecureQueryExecutor {
    async fn execute_secure_query(&self, query: SecureQuery) -> Result<Vec<PasswordEntry>> {
        // 1. 解析查询条件
        let conditions = self.parse_conditions(&query);

        // 2. 分离加密和非加密条件
        let (encrypted_conds, normal_conds) = self.separate_conditions(conditions);

        // 3. 执行非加密条件查询
        let candidate_ids = self.execute_normal_query(normal_conds).await?;

        // 4. 对候选结果应用加密条件过滤
        let filtered_ids = if encrypted_conds.is_empty() {
            candidate_ids
        } else {
            self.filter_encrypted_conditions(candidate_ids, encrypted_conds).await?
        };

        // 5. 安全地获取和解密结果
        let results = self.secure_fetch_and_decrypt(filtered_ids).await?;

        Ok(results)
    }

    async fn filter_encrypted_conditions(
        &self,
        candidate_ids: Vec<String>,
        conditions: Vec<EncryptedCondition>
    ) -> Result<Vec<String>> {
        let mut filtered_ids = Vec::new();

        // 分批处理，避免内存中同时存在太多解密数据
        for chunk in candidate_ids.chunks(50) {
            let entries = self.encryption_service
                .batch_decrypt(chunk)
                .await?;

            for entry in entries {
                if self.matches_encrypted_conditions(&entry, &conditions) {
                    filtered_ids.push(entry.id.clone());
                }

                // 立即清除内存中的解密数据
                self.encryption_service.secure_erase(&entry);
            }
        }

        Ok(filtered_ids)
    }
}
```
