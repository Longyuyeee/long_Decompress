### 3.1 格式检测服务
```python
class FormatDetector:
    """格式检测服务"""

    FORMAT_SIGNATURES = {
        b'PK\x03\x04': 'zip',
        b'Rar!\x1a\x07\x00': 'rar4',
        b'Rar!\x1a\x07\x01\x00': 'rar5',
        b'7z\xbc\xaf\x27\x1c': '7z',
        b'\x1f\x8b\x08': 'gz',
        b'BZh': 'bz2',
        b'\xfd7zXZ\x00': 'xz',
        b'ustar': 'tar',
    }

    def detect_format(self, filepath):
        """检测文件格式"""
        # 1. 检查扩展名
        ext = self._get_extension(filepath).lower()

        # 2. 读取文件头进行魔数检测
        with open(filepath, 'rb') as f:
            header = f.read(1024)

        # 3. 匹配魔数
        for magic, format_name in self.FORMAT_SIGNATURES.items():
            if header.startswith(magic):
                return format_name

        # 4. 回退到扩展名
        return self._extension_to_format(ext)

    def is_encrypted(self, filepath, format_name):
        """检测文件是否加密"""
        # 不同格式的加密检测逻辑
        if format_name == 'zip':
            return self._check_zip_encryption(filepath)
        elif format_name.startswith('rar'):
            return self._check_rar_encryption(filepath)
        elif format_name == '7z':
            return self._check_7z_encryption(filepath)
        return False
```

```python
from abc import ABC, abstractmethod

class ExtractEngine(ABC):
    """解压引擎抽象基类"""

    @abstractmethod
    def extract(self, source_path, dest_path, password=None):
        """解压文件"""
        pass

    @abstractmethod
    def list_contents(self, source_path, password=None):
        """列出压缩包内容"""
        pass

    @abstractmethod
    def test_integrity(self, source_path, password=None):
        """测试完整性"""
        pass

    @abstractmethod
    def get_info(self, source_path):
        """获取压缩包信息"""
        pass

    @property
    @abstractmethod
    def supported_formats(self):
        """支持的格式列表"""
        pass


class ZipEngine(ExtractEngine):
    """ZIP解压引擎"""

    def __init__(self):
        import zipfile
        self.zipfile = zipfile

    def extract(self, source_path, dest_path, password=None):
        try:
            with self.zipfile.ZipFile(source_path, 'r') as zip_ref:
                if password:
                    zip_ref.setpassword(password.encode())
                zip_ref.extractall(dest_path)
            return True
        except Exception as e:
            raise ExtractError(f"ZIP解压失败: {str(e)}")

    @property
    def supported_formats(self):
        return ['zip', 'zipx']
```

### 3.3 批量处理器
```python
import concurrent.futures
from typing import List, Dict, Callable

class BatchProcessor:
    """批量处理器"""

    def __init__(self, max_workers=None):
        self.max_workers = max_workers or min(4, os.cpu_count())
        self.results = {}
        self.errors = {}

    def process_batch(self, file_list: List[str],
                     output_dir: str,
                     config: Dict) -> Dict:
        """处理批量文件"""

        # 准备任务
        tasks = self._prepare_tasks(file_list, output_dir, config)

        # 创建进度跟踪
        progress = ProgressTracker(len(tasks))

        # 使用线程池执行
        with concurrent.futures.ThreadPoolExecutor(
            max_workers=self.max_workers) as executor:

            # 提交所有任务
            future_to_file = {
                executor.submit(self._process_single, task): task
                for task in tasks
            }

            # 收集结果
            for future in concurrent.futures.as_completed(future_to_file):
                task = future_to_file[future]
                try:
                    result = future.result()
                    self.results[task['source']] = result
                    progress.update_success()
                except Exception as e:
                    self.errors[task['source']] = str(e)
                    progress.update_failure()

        return {
            'total': len(file_list),
            'success': len(self.results),
            'failed': len(self.errors),
            'results': self.results,
            'errors': self.errors
        }

    def _process_single(self, task: Dict) -> Dict:
        """处理单个文件"""
        source = task['source']
        dest = task['dest']
        password = task.get('password')

        # 选择解压引擎
        engine = self._select_engine(source)

        # 执行解压
        start_time = time.time()
        engine.extract(source, dest, password)
        elapsed = time.time() - start_time

        return {
            'source': source,
            'destination': dest,
            'success': True,
            'elapsed_time': elapsed,
            'engine': engine.__class__.__name__
        }
```

### 3.4 密码管理器
```python
import json
from cryptography.fernet import Fernet

class PasswordManager:
    """密码管理器"""

    def __init__(self, config_dir):
        self.config_dir = config_dir
        self.passwords_file = os.path.join(config_dir, 'passwords.json')
        self.key_file = os.path.join(config_dir, 'key.key')
        self._init_encryption()

    def _init_encryption(self):
        """初始化加密"""
        if not os.path.exists(self.key_file):
            key = Fernet.generate_key()
            with open(self.key_file, 'wb') as f:
                f.write(key)

        with open(self.key_file, 'rb') as f:
            key = f.read()
        self.cipher = Fernet(key)

    def add_password(self, password: str, tags: List[str] = None):
        """添加密码"""
        passwords = self.load_passwords()

        # 加密密码
        encrypted = self.cipher.encrypt(password.encode()).decode()

        passwords.append({
            'password': encrypted,
            'tags': tags or [],
            'added': datetime.now().isoformat(),
            'used_count': 0,
            'last_used': None
        })

        self.save_passwords(passwords)

    def try_passwords(self, filepath: str,
                     max_attempts: int = 100) -> Optional[str]:
        """尝试密码列表"""
        passwords = self.load_passwords()

        # 智能排序：常用密码优先
        sorted_passwords = sorted(
            passwords,
            key=lambda x: (-x.get('used_count', 0), x.get('added', ''))
        )

        # 尝试密码
        for i, pwd_data in enumerate(sorted_passwords[:max_attempts]):
            try:
                password = self.cipher.decrypt(
                    pwd_data['password'].encode()
                ).decode()

                # 测试密码
                if self._test_password(filepath, password):
                    # 更新使用统计
                    pwd_data['used_count'] = pwd_data.get('used_count', 0) + 1
                    pwd_data['last_used'] = datetime.now().isoformat()
                    self.save_passwords(passwords)

                    return password

            except Exception:
                continue

        return None
```

### 3.5 错误处理器
```python
class ErrorHandler:
    """错误处理器"""

    ERROR_CATEGORIES = {
        'file_corrupted': {
            'user_message': "文件可能已损坏",
            'suggestions': [
                "尝试使用文件修复工具",
                "重新下载文件",
                "检查文件完整性"
            ],
            'recovery': 'skip'
        },
        'wrong_password': {
            'user_message': "密码不正确",
            'suggestions': [
                "尝试空密码",
                "尝试文件名作为密码",
                "使用密码本功能"
            ],
            'recovery': 'retry_with_different'
        },
        'disk_full': {
            'user_message': "磁盘空间不足",
            'suggestions': [
                "清理磁盘空间",
                "选择其他磁盘",
                "删除临时文件"
            ],
            'recovery': 'change_location'
        }
    }

    def handle_error(self, error: Exception, context: Dict) -> Dict:
        """处理错误"""
        # 分类错误
        error_type = self._classify_error(error)

        # 获取处理方案
        if error_type in self.ERROR_CATEGORIES:
            category = self.ERROR_CATEGORIES[error_type]
        else:
            category = self.ERROR_CATEGORIES['unknown']

        # 添加上下文信息
        response = {
            'type': error_type,
            'message': str(error),
            'user_message': category['user_message'],
            'suggestions': category['suggestions'],
            'recovery_action': category['recovery'],
            'context': context,
            'timestamp': datetime.now().isoformat()
        }

        # 记录错误
        self._log_error(response)

        return response

    def _classify_error(self, error: Exception) -> str:
        """错误分类"""
        error_str = str(error).lower()

        if any(word in error_str for word in ['corrupt', 'damage', 'bad']):
            return 'file_corrupted'
        elif any(word in error_str for word in ['password', 'encrypt']):
            return 'wrong_password'
        elif any(word in error_str for word in ['space', 'full', 'disk']):
            return 'disk_full'
        elif any(word in error_str for word in ['permission', 'access']):
            return 'permission_denied'
        else:
            return 'unknown'
```

```python
from PyQt5.QtCore import Qt, pyqtSignal
from PyQt5.QtGui import QDragEnterEvent, QDropEvent

class DropArea(QLabel):
    """拖放区域"""

    files_dropped = pyqtSignal(list)

    def __init__(self):
        super().__init__()
        self.setAlignment(Qt.AlignCenter)
        self.setText("将文件拖放到此处")
        self.setStyleSheet("""
            DropArea {
                border: 2px dashed #aaa;
                border-radius: 10px;
                background-color: #f8f9fa;
            }
            DropArea:hover {
                border-color: #007bff;
                background-color: #e9f5ff;
            }
        """)

    def dragEnterEvent(self, event: QDragEnterEvent):
        """拖入事件"""
        if event.mimeData().hasUrls():
            event.acceptProposedAction()
            self.setStyleSheet("""
                DropArea {
                    border: 2px dashed #007bff;
                    border-radius: 10px;
                    background-color: #e9f5ff;
                }
            """)

    def dragLeaveEvent(self, event):
        """拖出事件"""
        self.setStyleSheet("""
            DropArea {
                border: 2px dashed #aaa;
                border-radius: 10px;
                background-color: #f8f9fa;
            }
        """)

    def dropEvent(self, event: QDropEvent):
        """放下事件"""
        urls = event.mimeData().urls()
        files = [url.toLocalFile() for url in urls]

        self.files_dropped.emit(files)

        # 恢复样式
        self.setStyleSheet("""
            DropArea {
                border: 2px dashed #aaa;
                border-radius: 10px;
                background-color: #f8f9fa;
            }
        """)
```

```python
from logging.handlers import RotatingFileHandler

class StructuredLogger:
    """结构化日志记录器"""

    def __init__(self, log_dir):
        self.log_dir = log_dir
        self.setup_logging()

    def setup_logging(self):
        """设置日志"""
        # 创建日志目录
        os.makedirs(self.log_dir, exist_ok=True)

        # 配置根日志记录器
        logger = logging.getLogger('long_decompress')
        logger.setLevel(logging.INFO)

        # 文件处理器（按大小轮转）
        file_handler = RotatingFileHandler(
            os.path.join(self.log_dir, 'app.log'),
            maxBytes=10*1024*1024,  # 10MB
            backupCount=5,
            encoding='utf-8'
        )

        # 控制台处理器
        console_handler = logging.StreamHandler()

        # 格式化器
        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        file_handler.setFormatter(formatter)
        console_handler.setFormatter(formatter)

        # 添加处理器
        logger.addHandler(file_handler)
        logger.addHandler(console_handler)

        self.logger = logger

    def log_extraction(self, source: str, dest: str,
                      success: bool, details: Dict):
        """记录解压操作"""
        log_entry = {
            'action': 'extraction',
            'timestamp': datetime.now().isoformat(),
            'source': source,
            'destination': dest,
            'success': success,
            'details': details
        }

        if success:
            self.logger.info(json.dumps(log_entry, ensure_ascii=False))
        else:
            self.logger.error(json.dumps(log_entry, ensure_ascii=False))
```
