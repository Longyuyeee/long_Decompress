## 4. 智能默认设置

### 4.1 解压路径智能选择
```python
def smart_extract_path(source_file):
    """
    智能选择解压路径算法
    """
    source_dir = os.path.dirname(source_file)
    filename = os.path.basename(source_file)
    name_without_ext = os.path.splitext(filename)[0]

    # 规则1：检查是否存在同名解压文件夹
    default_path = os.path.join(source_dir, f"{name_without_ext}_extracted")

    # 规则2：避免覆盖，自动递增编号
    counter = 1
    while os.path.exists(default_path):
        default_path = os.path.join(source_dir, f"{name_without_ext}_extracted_{counter}")
        counter += 1

    # 规则3：检查磁盘空间
    if not has_enough_space(source_file, default_path):
        # 建议其他有空间的磁盘
        return suggest_alternative_drive()

    return default_path
```

### 4.2 冲突处理默认策略
- **文件已存在**: 自动重命名（添加_1, _2等后缀）
- **文件夹已存在**: 合并内容，冲突文件重命名
- **权限不足**: 尝试当前目录，不行则建议用户目录
- **空间不足**: 自动选择其他有空间的磁盘

### 4.3 密码处理智能默认
- **首次遇到加密文件**: 显示简洁密码输入框
- **密码正确后**: 询问"是否记住此密码用于类似文件？"
- **密码本发现**: 自动扫描同目录下的password.txt, passwords.txt等文件

class BatchConfigSmartSuggest:
    def analyze_files(self, file_list):
        """分析文件列表，给出智能建议"""
        stats = {
            'total_files': len(file_list),
            'total_size': sum(os.path.getsize(f) for f in file_list),
            'formats': self._count_formats(file_list),
            'encrypted_count': self._count_encrypted(file_list),
            'password_files': self._find_password_files(file_list)
        }

        return self._generate_suggestions(stats)

    def _generate_suggestions(self, stats):
        """根据统计生成建议"""
        suggestions = {}

        # 解压路径建议
        if stats['total_files'] > 10:
            suggestions['output_path'] = f"批量解压_{datetime.now():%Y%m%d_%H%M%S}"
        else:
            suggestions['output_path'] = "解压结果"

        # 处理模式建议
        if stats['encrypted_count'] > 0:
            if stats['password_files']:
                suggestions['password_mode'] = 'auto_from_files'
            else:
                suggestions['password_mode'] = 'ask_first_then_skip'

        # 线程数建议（基于CPU核心和文件数）
        cpu_cores = os.cpu_count()
        suggestions['threads'] = min(cpu_cores, max(1, stats['total_files'] // 5))

        return suggestions
```

def user_friendly_extract(file_path, output_dir=None):
    """
    用户友好的解压函数，包含完整错误处理
    """
    try:
        # 1. 验证输入
        validate_input_file(file_path)

        # 2. 准备输出目录
        output_dir = output_dir or smart_extract_path(file_path)
        prepare_output_directory(output_dir)

        # 3. 执行解压
        result = extract_with_progress(file_path, output_dir)

        # 4. 验证结果
        verify_extraction(result)

        # 5. 清理和反馈
        cleanup_temp_files()
        show_success_message(result)

        return result

    except ExtractError as e:
        # 用户友好的错误处理
        error_info = classify_error(e)
        user_message = format_user_message(error_info)
        recovery_options = suggest_recovery(error_info)

        show_error_dialog(user_message, recovery_options)

        # 记录错误用于改进
        log_error_for_improvement(e, context)

        return None
```
